# Configuração do HTTPS
server {
    listen 443 ssl;
    server_name licitacao360.com www.licitacao360.com;

    # Diretório do frontend (se houver)
    location / {
        root /var/www/licitacao360;
        index index.html;
        try_files $uri /index.html;
    }

    # Configuração do backend API
    location /api/ {
        include proxy_params;
        proxy_pass http://unix:/run/licitacao360.sock;

    }

    # Logs de acesso e erro para o domínio
    access_log /var/log/nginx/licitacao360_access.log;
    error_log /var/log/nginx/licitacao360_error.log;
}
